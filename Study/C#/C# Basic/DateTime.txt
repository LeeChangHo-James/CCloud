[DateTime]
● DateTime 
  ▶ 특정시간입력
     ▶ DateTime 변수 = new DateTime(int a, int b, int c, int d, int e, int f);
        a = 년 , b = 월 , c = 일 , d = 시, e = 분, f = 초
  ▶ 현재시간입력
     ▶ DateTime 변수 = DateTime.Now;

● DateTime의 값 출력하기
  ▶ DateTime 변수 = new DateTime();
     변수 = DateTime.Now;
<Unity>
      Text text1 = 변수.ToString("yyyy-MM-dd HH:mm:s");
      Text text2 = 변수.ToString("yyyy=mm=dd");
      Text text3 = 변수.ToString("HH:mm=dd);

  ▶ 시간 입력
     ▶ 년
        ▶ y
           ▶ 한 자리 연도, 2001은 "1"로 표시된다.
        ▶ yy
           ▶ 연도의 마지막 두 자리, 2001은 "01"로 표시
        ▶ yyyy
           ▶ 완전한 형태의 연도, 2001은 "2001"로 표시
     ▶ 월
        ▶ M
           ▶ 달을 나타내는 한 자리 또는 두자리 숫자
        ▶ MM
           ▶ 달을 나타내는 두 자리 숫자, 한 자리로 된 값 앞에는 0이 온다
        ▶ MMM
           ▶ 세 문자로 된 달의 약어
        ▶ MMMM
           ▶ 달의 전체 이름
     ▶ 일
        ▶ d
           ▶ 한 자리 또는 두 자리 날짜
        ▶ dd
           ▶ 두 자리 날짜, 한 자리로 된 날짜 값 앞에는 0이 온다.
     ▶ 시간
        ▶ h
           ▶ 12시간 형식의 한 자리 또는 두 자리 시간
        ▶ hh
           ▶ 12시간 형식의 두 자리 시간, 한 자리로 된 값 앞에는 0이 온다
        ▶ H
           ▶ 24시간 형식의 한 자리 또는 두 자리 시간
        ▶ HH
           ▶ 24시간 형식의 두 자리 시간, 한 자리로 된 값 앞에는 0이 온다
     ▶ 분
        ▶ m
           ▶ 한 자리 또는 두 자리 분
        ▶ mm
           ▶ 두 자리 분, 한 자리로 된 값 앞에는 0이 온다
     ▶ 초
        ▶ s
           ▶ 한 자리 또는 두 자리 초
        ▶ ss
           ▶ 두 자리 초, 한 자리로 된 값 앞에는 0이 온다
     ▶ Am/Pm
        ▶ t
           ▶ 한 문자로 된 A.M/P.M 약어 A.M.은 "A"로 표시된다
        ▶ tt
           ▶ 두 문자로 된 A.M/P.M. 약어, A.M.은 "AM"으로 표시된다.
