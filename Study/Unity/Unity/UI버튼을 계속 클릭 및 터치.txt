[UI버튼을 계속 클릭 / 터치가 해야 할 경우]
● IPointerDownHandler, IPointerUpHandlder 인터페이스
  ▶ 유니티가 제공하는 인터페이스 중에서 IPointerDownHandler, IPointerUpHandlder 인터페이스를 사용함으로서 버튼이 누르고 있을 때만 특정 기능 수행이 가능하다.
    ▶ public void OnPointerDown(PointerEventData eventData) : 버튼을 클릭/터치하는 순간 실행
    ▶ public void OnPointerUp(PointerEventData eventDate) : 버튼 클릭/터치를 떼는 순간 실행
       ▶ ex)
          using System.Collections;
          using System.Collections.Generic;
          using UnityEngine;
          using UnityEngine.EventSystems;

          public class BtnCtrl : MonoBehaviour, IPointerDownHandler, IPointerUpHandlder {

              private bool isBtnDown = false;

              private void Update()
              {
                  if (isBtnDown) {
                      Debug.Log ("BTN DOWN"); // 버튼을 누르고 있을 때만 특정 기능을 프레임마다 반복해서 실행
                  }
              }

              public void OnPointerDown (PointerEventData eventData)
              {
                  isBtnDown = true;
              }

              public void OnPointerUp (PointerEventData eventData)
              {
                  isBtnDown = false;
              }
          }
       ▶ OnPointerDown() 메소드 자체는 프레임마다 반복해서 발동하지 않는다. 오직 1번만 발동
          ∴ bool 변수와 Update() 메소드를 함께 사용해주어야 한다.
    ▶ 유니터 에디터에서 EventTrigger 컴포넌트를 추가해서 구현하는 것도 가능.
