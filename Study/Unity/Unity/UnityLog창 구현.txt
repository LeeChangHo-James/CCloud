  ▶ ScrollView → Content 안에 Text UI를 추가
  ▶ ScrollView 안에 "Content Size Fitter" , "Vertical Layout Group" 추가
     ● Contetn Size Fitter
       ▶ Horizontal Fit = Preferred Size
       ▶ Vertical Fit = Preferred Size
     ● Vertical Layout Group
       ▶ Padding
       ▶ Spacing = 0
       ▶ Child Alignment Upper Left
       ▶ Control Child Size
          ▶ Width 체크
          ▶ Height 체크
       ▶ Child Force Expand
          ▶ Width 체크
          ▶ Height 체크
   ▶ ex)
      using UnityEngind;
      using UnityEngind.UI;

      public class mLogWindow : MonoBehaviour {
          
          private Text logText= null; // 로그를 보여줄 Text UI
          private ScrollRect scroll_rect = null; // 스크롤바

          void Start() {
              logText = GameObject.Find("log_Text").GetComponent <Text> ();
              scroll_rect = GameObject.Find("Scroll_View").GetComponent<ScrollRect>();

              if (logText != null)
              logText.text += "Hello Log Window!" + "\n";
          }

          void Update() {

              if (Input.GetMouseButtonDown(0))
              {
                  logText.text += "Mouse down position (" + "X : " + Input.mousePosition.x + "Y : " 
                  + Input.mousePosition.y + ")\n"; // 현재 마우스의 위치를 TextUI에 추가해준다.

                  // 스크롤바의 위치를 제일 아래로 내려준다.
                  // 1.0이면 제일 위로 스크롤 0.0 이면 제일 아래로 스크롤이다.
                  scroll_rect.VerticalNormalizedPosition = 0.0f;
              }
          }
      }
